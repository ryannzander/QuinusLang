Advanced Features

Type Aliases

Create type aliases with alias for readability:

alias Id = u64;
alias Buffer = [u8; 256];
alias FloatPair = (f64, f64);

make x: Id = 42;
make buf: Buffer = { 0 };

Defer Blocks

Run cleanup code when leaving the current scope with defer:

{
    make shift f = open_file("data.txt");
    defer {
        close(f);
    }
    // ... use f ...
}  // close(f) runs here

Range Iteration

Iterate over numeric ranges with foreach ... in start..end:

foreach i in 0..10 {
    print(i);
}

foreach n in 5..15 {
    print(n * 2);
}

The range is exclusive of the end value (0..10 iterates 0 through 9).

Tuples and Multiple Return Values

Return multiple values using tuple types:

craft div_rem(a: i32, b: i32) -> (i32, i32) {
    send (a / b, a % b);
}

craft main() -> void {
    make (q, r) = div_rem(17, 5);
    print(q);  // 3
    print(r);  // 2
    send;
}

Tuple literals use parentheses: (1, 2, 3).

Default Parameters

Parameters can have default values:

craft greet(name: str, count: i32 = 1) -> void {
    foreach i in 0..count {
        print(name);
    }
    send;
}

greet("Hi");      // count defaults to 1
greet("Hi", 3);   // count is 3

Pattern Matching: choose

Match on enum values with choose:

state Color {
    Red,
    Green,
    Blue
}

craft describe(c: Color) -> void {
    choose (c) {
        Red => { print("red"); }
        Green => { print("green"); }
        Blue => { print("blue"); }
        _ => { print("unknown"); }
    }
    send;
}

Enum Variants with Payloads

Enums can carry data in their variants:

state Maybe {
    None,
    Some(i32)
}

craft unwrap(m: Maybe) -> i32 {
    choose (m) {
        Some(x) => { send x; }
        None => { panic(); send 0; }
    }
}

Method Syntax for Structs

Add methods to structs with impl blocks:

form Point {
    x: i32,
    y: i32
}

impl Point {
    craft distance(self: Point) -> f64 {
        make dx: f64 = self.x;
        make dy: f64 = self.y;
        send sqrt(dx*dx + dy*dy);
    }
}

craft main() -> void {
    make shift p: Point = { x: 3, y: 4 };
    print(p.distance());
    send;
}

Methods take self as the first parameter. Call with value.method().
